








A key insight is that sampling from the backward transition kernel~\eqref{eq:backward_kernel}, given access to score estimate $\scf_t$, is a special case of the setup of Section~\ref{sec:gaussian_tilts}, except that we only have approximately correct first-order evaluations.
This leads to the template~\cref{alg:DM} for high-accuracy diffusion sampling.
In the following subsections, we show how to choose $q_t$ and $\mathcal W_x^t$ for $t\in [T]$.

\renewcommand{\xr}[1]{X_{#1}}
\begin{algorithm}[H]
  \caption{Backward diffusion sampling}\label{alg:DM}
\begin{algorithmic}
\STATE \textbf{Input:} Score estimates $\crl{\scf_t}_{t\in[T]}$, initial distribution $\phat_T$, parameters $(\alp_t,\eta_t)_{t\in [T]}$
\STATE Sample $\xr{T}\sim \phat_T$.
\FOR{$t=T-1,\dotsc,1$}
\STATE Sample $\xr{t}\leftarrow \AlgWSC(B, q_t, (\mathcal W^t_x)_{x\in\R^d})$.
\ENDFOR
\STATE \textbf{Output:} $\xr{1}$
\end{algorithmic}
\end{algorithm}








\subsection{A simple method}\label{ssec:diffusion-simple}


We first propose a simple method that uses as a proposal the Gaussian portion of~\eqref{eq:backward_kernel}.
We state all our guarantees for sampling from the early stopped distribution $p_1$, the law of $X_1 \sim \normal{\alpha_0 X_0, \sigma_1^2}$ with $X_0 \sim \pdata$; see the discussion after the theorem.

\begin{theorem}\label{thm:DM-clip-simple}
Suppose that $\delta\in(0,\frac12]$, $\B=\Theta(1)$, and
\begin{align}\label{eq:DM-simple-eta}
    \frac{\sigma_t^2}{\eta_{t}} \gg d\log(1/\delta)+\log^2(1/\delta)\,.
\end{align}

Consider instantiating the subroutine
$\AlgWSC$ in \cref{alg:DM} with the following choices: $q_t = \nor(\bar x_t, \eta_t I)$, where $\bar x_t \deq \alpha_t^{-1} \xr{t+1}$, and $\mathcal W_x^t$ is the law of
\begin{align*}
    \hW_{r,x} \deq \pclip{\langle x-\bar x_t, \scf_t(rx+(1-r)\bar x_t)\rangle}\,,
\end{align*}
for $r\sim \Unif([0,1])$.

Let $\phat_1$ be the law of $\xr{1}$ generated by \cref{alg:DM}. Then,
\begin{align*}
\kl{p_1}{\phat_1}\leqsim&~ \kl{p_T}{\phat_T}+T\delta+ \sum_{t=1}^{T-1} \eta_t \epssct^2\,.
\end{align*}
\end{theorem}

Next, we describe the implications of~\cref{thm:DM-clip-simple} for diffusion model sampling.
We should always choose $\B=\Theta(1)$ %
and we denote $G\deq C(d+\log(1/\delta))\log(d/\delta)$ for a sufficiently large constant $C$ from \cref{eq:DM-simple-eta}.
Then, in the variance-preserving setting:%
\begin{itemize}
    \item \scedit{Recall that} $\alp_t^2=\frac{1-\sigma_{t+1}^2}{1-\sigma_t^2}$ and $\sigma_{t+1}^2=\frac{1-\sigma_{t+1}^2}{1-\sigma_t^2}\,(\sigma_{t}^2+\eta_{t})$.
    \item The condition $\eta_t\leq \frac{\sigma_t^2}{G}$ reduces to $\frac{\sigma_{t+1}^2}{1-\sigma_{t+1}^2}\leq \frac{\sigma_t^2}{1-\sigma_t^2}\cdot \prn*{1+\frac{1}{G}}$. 
    \item This implies that as long as $T\geq O(G\log(1/(\bar\delta \sigma_0^2)))$, we can guarantee that $1-\sigma_T^2\leq \bar\delta$.
    \item From known results on the convergence of the forward process, $1-\sigma_T^2 \le \bar\delta$ and $\phat_T=\normal{0,\sigma_T^2\Id}$ imply that $\kl{p_T}{\phat_T}\lesssim \bar\delta \En\nrm{X_0}^2$.

\end{itemize}
By the above calculation, we can show the following corollary. We denote $\mathsf M_2^2 \deq \E_{X_0\sim \pdata}\|X_0\|^2$ to be the second moment of $\pdata$.
\begin{corollary}\label{cor:poly-log}
In the variance-preserving setting, for any $\delta\in(0,\frac12]$, $\sigma_0>0$, there exists a schedule $(\sigma_t)_{t\in[T]}$ such that
\begin{align*}
    T\leq O\prn[\big]{(d+\log(1/\delta))\log^2(\kappa/\delta)}\,,
\end{align*}
where $\kappa\deq (d+M_2^2)/\sigma_0^2$, 
and \cref{alg:DM} can be instantiated with $\phat_T=\normal{0,\sigma_T^2\Id}$, so that
\begin{align*}
    \kl{p_1}{\phat_1}\leqsim&~ \delta^2+\sum_{t=1}^{T-1} \eta_t \epssct^2\,.
\end{align*}
Further, the number of queries made by~\cref{alg:DM} is $O(T)$ \whp.
\end{corollary}
    This governs the complexity of sampling from the early stopped distribution $p_1$.
    To convert this into a guarantee for sampling from $\pdata$ itself, we can note that $W_2^2(\pdata, p_1)\le (1-\alpha_0)^2\,\E\|X_0\|^2 + \sigma_0^2 d$.
    This can be made at most $\delta^2$ by choosing $\sigma_0^2 \asymp \delta^2/(d+\mathsf M_2^2)$.

In terms of the bounded Lipschitz metric, this implies that $\Dbl{\pdata}{\widehat p_1}^2 \lesssim \delta^2 + \sum_{t=1}^{T-1} \eta_t \epssct^2$ with a total complexity of %
\begin{align*}
    \boxed{\max\{d, \log(1/\delta)\} \log^2\prn[\Big]{\frac{d+\mathsf M_2^2}{\delta^2}}}.
\end{align*}
Note that this guarantee imposes no assumptions on $\pdata$ beyond a second moment bound.
This improves upon prior works~\citet{Ben+24Diffusion, ConDurSil25KLDiffusion} which achieved $\widetilde O(d/\delta^2)$ under these assumptions, and~\citet{JaiZha25SharpKL, LiYan25Diffusion} which recently improved the complexity to $\widetilde O(d/\delta)$.

 






\subsection{A DDPM-like method}


Next, we present a method which we call a ``DDPM-like'' method because the proposal $q$ takes the form $\normal{\alpha_t^{-1} \xr{t+1} + \eta_t \scf_t(\alpha_t^{-1} \xr{t+1}), \eta_t I}$, which is similar but slightly different %
from a step of DDPM~\eqref{eq:DDPM-approx}.


We show that our method, similar to the one from~\cref{sec:gaussian_tilts}, achieves $\sqrt{d}$-complexity under the \emph{non-uniform Lipschitz condition}, following~\citet{jiao2024instance, jiao2025optimal}.

\newcommand{\Ldel}{L_\delta}
\newcommand{\piz}[2][\xz]{\nu_{#1}(#2)}

\begin{assumption}[Non-uniform Lipschitz condition]\label{asmp:non-unif-Lip}
For any $\delta>0$, there is a parameter $\Ldel\geq 1$ such that for any $t\in[T]$,
\begin{align}
    \bbP_{X_t\sim p_t}\prn[\Big]{ \nrmop{\nabla \scfs_t(X_t)}> \frac{\Ldel}{\sigma_t^2} }\leq \frac{\delta^2}{d^8}\,.
\end{align}
\end{assumption}

We note that by \cref{prop:nabla-p-MGF}, \cref{asmp:non-unif-Lip} holds unconditionally with $L_\delta=O(d+\log(1/\delta))$. In other words, the score function $\scfs_t$ is smooth ``with high probability''. Further, when $\pdata$ is Gaussian, \cref{asmp:non-unif-Lip} holds with $L_\delta\equiv 1$ for any $\delta\geq 0$. 
As shown in~\citet{jiao2025optimal}, when $\pdata=\sum_{h=1}^H p_h\normal{\mu_h,\sigma_h^2}$ is a mixture of $H$ Gaussian distributions, it holds that $L_\delta\leq O(\log(dH/\delta))$ (see also \cref{prop:logp-adapt}).






\begin{theorem}\label{thm:DM-clip-path}
Suppose that \cref{asmp:non-unif-Lip} holds, $\delta\in(0,\frac12]$, $\B=\Theta(1)$, and
\begin{align}
    \frac{\sigma_t^2}{\eta_{t}}\gg \sqrt{dL_\delta\log(d/\delta)}+L_\delta\log(d/\delta)\,.
\end{align}

Consider instantiating the subroutine $\AlgWSC$ in \cref{alg:DM} with the following choices: $q_t = \normal{\bar x_t + \eta_t \scf_t(\bar x_t), \eta_t I}$, where $\bar x_t \deq \alpha_t^{-1} \xr{t+1}$, and $\mc W_x^t$ is the law of
\begin{align*}
    \widehat W_{r,z,x} \deq \pclip{\langle \dot \gamma_{z,r,\bar x_t}(x), \scf_t(\gamma_{z,r,\bar x_t}(x)) - \scf_t(\bar x_t)\rangle}\,,
\end{align*}
where $z\sim \normal{0,\eta_t I}$, $r\sim\Unif([0,1])$, 
and $\gamma_{z,r,\bar x_t}$ denotes the path function
\begin{align}
    \gamma_{z,r,\bar x_t}(x) \deq\alr x + (1-\alr) \bar x_t + \blr z\,, \label{eq:path_fn_1} \\
    \alr\deq\sin(\pi\lr/2)\,, ~~\blr\deq\cos(\pi\lr/2)\,.\label{eq:path_fn_2}
\end{align}
Let $\phat_1$ be the law of $\xr{1}$ generated by \cref{alg:DM}. Then
\begin{align*}
    \MoveEqLeft\Dtv{p_1}{\phat_1}^2\leq 2\Dhels{p_1}{\phat_1} \icml{\\
    &}\leqsim \kl{p_T}{\phat_T}+T\delta+
    \sqrt{d/L_\delta}\cdot\sum_{t=1}^{T-1} \eta_t \epssct^2\,.
\end{align*}
\end{theorem}

Following the reasoning of the preceding subsection, \cref{thm:DM-clip-path} implies a complexity of
\begin{align*}
    \boxed{\max\{\sqrt{dL_\delta\log(d/\delta)}, L_\delta \log(d/\delta)\} \log \frac{d+\mathsf M_2^2}{\delta^2}}
\end{align*}
to reach $\Dbl{\wh p_1}{\pdata}^2 \lesssim \delta^2 + \sqrt{d/L_\delta} \sum_{t=1}^{T-1} \eta_t \epssct^2$.
Note that the dependence on the score error is worse; we do not know if this is fundamental.


\subsection{Adaptation to low-dimensional structure}

\newcommand{\tscf}{\wt{\scf}}
\newcommand{\tscfs}{\wt{\scf}^\star}

While \cref{thm:DM-clip-path} improves the dependence on the ambient dimension to $\sqrt d$, ideally, we would like to choose $\eta$ according to the \emph{intrinsic dimension} of the data distribution $\pdata$~\citep{li2024adapting} (cf. \cref{def:intrinsic}).

The key observation is that for $p=\pdata*\Nsigma$, we can write, for $X_0 \sim \pdata$, $X \sim \normal{X_0,\sigma^2 I}$,
\begin{align}
    \nabla^2 \log p(x)=-\frac{\Id}{\sigma^2}+
    \frac{\Cov(X_0\mid X=x)}{\sigma^4}\,,
\end{align}
and hence, when $\pdata$ has low-dimensional structure, we may expect the shifted Hessian $\nabla^2 \log p+\sigma^{-2}\Id$ to be well-behaved. This motivates adapting \cref{thm:DM-clip-path} by replacing the score function with the shifted score function $\tscf(x)=\scf(x)+\sigma^{-2} x$. 

\begin{definition}[Intrinsic dimension]\label{def:intrinsic}
Suppose that the data distribution $\pdata$ has support $\cXbar=\supp(\pdata)$. Let $\Ndata(r)=N(\cXbar; r, \nrm{\cdot}_2)$ be the covering number of $\cX$ under Euclidean norm, and we define the \emph{intrinsic dimension} of $\pdata$ as $\dstar=1+\log \Ndata(\sigma_0/\sqrt{d})$. 
\end{definition}

\begin{theorem}\label{thm:DM-clip-path-adapt}
Suppose that $\delta\in(0,\frac12]$, $\B=\Theta(1)$, and
\begin{align}
    \frac{\sigma_t^2}{\eta_{t}} \gg \dstar\log(d/\delta)+\log^2(d/\delta)\,.
\end{align}

Consider instantiating the subroutine $\AlgWSC$ in \cref{alg:DM} with the following choices: $q_t = \normal{\bar x_t + \bar\eta_t \scf_t(\bar x_t), \bar\eta_t I}$, where $\bar x_t \deq \alpha_t^{-1} \xr{t+1}$, $\lambda_t=\frac{1}{\sigma_t^2}$, $1/\bar\eta_t = 1/\eta_t + \lambda_t$, and $\mc W_x^t$ is the law of
\begin{align*}
    \widehat W_{r,z,x} \deq \mathsf{Clip}_B\bigl(\langle \dot \gamma_{z,r,\bar x_t}(x),&~\scf_t(\gamma_{z,r,\bar x_t}(x)) - \scf_t(\bar x_t) \\
    &\quad + \lambda_t\,(\gamma_{z,r,\bar x_t}(x) - \bar x_t)\rangle\bigr)\,,
\end{align*}
where $z\sim \normal{0,\eta_t I}$, $r\sim\Unif([0,1])$, 
and $\gamma_{z,r,\bar x_t}$ is the path function given in~\eqref{eq:path_fn_1}--\eqref{eq:path_fn_2}.

Let $\phat_1$ be the law of $\xr{1}$ generated by \cref{alg:DM}. Then
\begin{align*}
    \Dkl{p_1}{\phat_1}\leqsim&~ \kl{p_T}{\phat_T}+T\delta 
    +
    \frac{d}{\dstar}\sum_{t=1}^{T-1} \eta_t \bepssct^2\,,
\end{align*}
where for each $t\in[T]$, we define
\begin{align*}
    \bepssct[t]^2\ldef \max_{\eta\in[0,2\eta_t]}\En_{X\sim p_t*\normal{0,\eta\Id}}\nrm{\scf_t(X_t)-\scfs_t(X_t)}^2\,.
\end{align*}
\end{theorem}
This implies a complexity of
\begin{align*}
    \boxed{\max\{\dstar, \log(d/\delta)\} \log^2\prn[\Big]{ \frac{d+\mathsf M_2^2}{\delta^2}}}
\end{align*}
to reach $\Dbl{\pdata}{\wh p_1}^2 \lesssim \delta^2 + (d/\dstar) \sum_{t=1}^{T-1} \eta_t \bepssct^2$.
Again, the dependence on the score error is worse.
