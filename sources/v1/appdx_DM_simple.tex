\subsection{Proof of Theorem~\ref{thm:DM-clip-simple}}



\begin{proof}[\pfref{thm:DM-clip-simple}]
The process $\xr{T}\to\cdots\to\xr{1}$ generated by \cref{alg:DM} is a Markov chain such that $\xr{T}\sim \phat_T$, and $\xr{t}\mid \xr{t+1}\sim \wh\rho_t(\cdot\mid \xr{t+1})$.
The chain rule for the KL divergence yields
\begin{align*}
    \Dkl{p_1}{\wh p_1}
    &= \Dkl{p_T}{\wh p_T} + \sum_{t=1}^{T-1} \E_{\xr{t+1} \sim p_{t+1}}\Dkl{\rho_t(\cdot \mid \xr{t+1})}{\wh\rho_t(\cdot \mid \xr{t+1})}\,.
\end{align*}

Given $\xr{t+1} = x_{t+1}$ and $\bar x_t = \alpha_t^{-1} x_{t+1}$, \cref{alg:fors} is instantiated with base measure $q_t=\normal{\bar x_t,\eta_t\Id}$. Consider the target $\rho_t(\cdot \mid x_{t+1})$, and by~\cref{thm:fors},
\begin{align*}
    &~\log \wh\rho_t(x\mid x_{t+1}) - \log \rho_t(x\mid x_{t+1})-\const \\
    =&~ \E_{r\sim \Unif([0,1])} \pclip{\langle x-\bar x_t, \scf_t(rx+(1-r)\bar x_t)\rangle} - \log p_t(x)\\
    =&~ \E_{r\sim \Unif([0,1])}\brk[\big]{\pclip{\langle x-\bar x_t, \scf_t(rx+(1-r)\bar x_t)\rangle} - \langle x-\bar x_t,\nabla \log p_t(rx + (1-r)\bar x_t)\rangle}\,,
\end{align*}
and hence
\begin{align*}
    \abs{\log \wh\rho_t(x\mid x_{t+1}) - \log \rho_t(x\mid x_{t+1})-\const}\leq&~ \underbrace{\E_{r\sim\Unif([0,1])}\pclip{\abs{\langle x-\bar x_t, [\scf_t - \nabla \log p_t](rx+(1-r)\bar x_t)\rangle}}}_{\eqqcolon u(x)}  \\
    &~ + {\underbrace{\E_{r\sim \Unif([0,1])}\trunc{\langle x-\bar x_t, \nabla \log p_t(rx+(1-r)\bar x_t)\rangle}}_{\eqqcolon v(x)}},
\end{align*}
where we used $\abs{a-\pclip{b}}\leq \pclip{\abs{a-b}}+\trunc{b}$ for $a,b\in\RR$.
Applying \cref{lem:KL-decomp} gives
\begin{align*}
    \Dkl{\rho_t(\cdot\mid x_{t+1})}{\wh\rho_t(\cdot\mid x_{t+1})}
    \leq&~ e^\B\En_{x\sim \rho_t(\cdot\mid \bar x_t),\, \lr\sim \Unif([0,1])}\brk*{ \tri{x-\bar x_t,[\nabla \log p_t-\scf_t](\lr x+(1-\lr)\bar x_t)}^2\wedge \B^2 } \\
    &~+\En_{x\sim \rho_t(\cdot\mid \bar x_t),\, \lr\sim \Unif([0,1])}\prn*{ \exp\prn*{\tri{x-x',\nabla \log p_t(\lr x+(1-\lr)\bar x_t)}} - e^\B }_+\,.
\end{align*}
Taking expectation over $\xr{t+1}\sim p_{t+1}$ and using the fact that $\xr{t+1}\mid \xr{t}\sim \normal{\alpha_t\xr{t},\alpha_t^2\eta_t\Id}$, we can conclude that
\begin{align*}
    &~\En_{\xr{t+1}\sim p_{t+1}} \kl{\rho_t(\cdot\mid \xr{t+1})}{\wh\rho_t(\cdot\mid \xr{t+1})} \\
    \leq&~ e^\B\En_{\xr{t}\sim p_t,\, \lr\sim \Unif([0,1]),\,Z\sim \normal{0,\eta_t\Id}}\brk*{ \tri{Z,[\nabla \log p_t-\scf_t](\xr{t}+\lr Z)}^2\wedge \B^2 } \\
    &~+\En_{\xr{t}\sim p_t,\, \lr\sim \Unif([0,1]),\,Z\sim \normal{0,\eta_t\Id}}\prn*{ \exp\prn*{\abs{\tri{Z,\nabla \log p_t(\xr{t}+\lr Z)}}} - e^\B }_+\,.
\end{align*}
By \cref{prop:Gaussian-change-of-measure} (1), it holds that for any $\lr\in[0,1]$,
\begin{align*}
    &~\En_{\xr{t}\sim p_t,\, Z\sim \normal{0,\eta_t\Id}}\brk*{ \tri{Z,[\nabla \log p_t-\scf_t](\xr{t}+\lr Z)}^2\wedge \B^2 } \\
    \leq&~ 2\En_{\xr{t}\sim p_t,\,Z\sim \normal{0,\eta_t\Id}}\brk*{ \tri{Z,[\nabla \log p_t-\scf_t](\xr{t})}^2\wedge \B^2 }+O(\B^2)\cdot e^{-K/12} \\
    \leq&~ 2\eta_t \En_{\xr{t}\sim p_t}\nrm{\nabla \log p_t(\xr{t})-\scf_t(\xr{t})}^2+O(\B^2)\cdot e^{-K/12}\,,
\end{align*}
where we use $\En_{Z\sim \normal{0,\eta_t\Id}}[ZZ^\top]=\eta_t\Id$. 

By \cref{prop:Gaussian-change-of-measure} (2) and the fact that $\En[Y\indic\crl{Y>M}]\leq M^{\lambda-1}\En[Y^\lambda]$ for $\lambda>1$, it holds that for any $\lr\in[0,1]$,
\begin{align*}
    \En_{\xr{t}\sim p_t, Z\sim \normal{0,\eta_t\Id}}\prn*{ \exp\prn*{\abs{\tri{Z,\nabla \log p_t(\xr{t}+\lr Z)}}} - e^\B }_+\leq O(e^\B)\cdot e^{-c_\B K}.
\end{align*}
Combining these inequalities completes the proof.
\end{proof}

\begin{proposition}\label{prop:Gaussian-change-of-measure}
Suppose that $\eta\leq \frac{\sigma^2}{\max\crl{Kd,K^2}}$ and $K\geq 2$, and $P=\bar P*\Nsigma$.


(1) For any $0\leq F\leq 1$ and $r\in[0,1]$, it holds that
\begin{align*}
    \En_{X\sim P,\, Z\sim \normal{0,\eta\Id}} F(Z,X+rZ)
    -2\En_{X\sim P,\, Z\sim \normal{0,\eta\Id}} F(Z,X)\leq 2\exp\prn[\Big]{-\frac{K}{12}}\,.
\end{align*}

(2) For any $r\in[0,1]$ and $\B>0$, it holds that
\begin{align*}
    \En_{X\sim P,\, Z\sim \normal{0,\eta\Id}} \prn*{\exp\prn*{ \abs*{\tri{Z, \nabla \log p(X+r Z)}} } -e^\B}_+ \leq e^{\B+1}\cdot \exp\prn[\Big]{-\frac{\min\crl{\B,1}\B}{32}K}\,.
\end{align*}


\end{proposition}

\begin{proof}
\newcommand{\Qr}[1][r]{Q^{#1}_{X_0}}
\newcommand{\bQr}[1][r]{\bQ^{#1}}
(1)
Fix any $X_0$, and let $\Qr$ be the distribution of $(W,Z)$ under $X\sim \normal{X_0,\sigma^2\Id}$, $Z\sim \normal{0,\eta\Id}$, and $W=X+r Z$. In other words, $\Qr=\normal{[X_0;\bfz], \Sigma_r}$ where $\Sigma_r=\begin{bmatrix}
    (\sigma^2+r^2\eta)\Id & r \eta\Id \\
    r \eta\Id & \eta\Id 
\end{bmatrix} $. 
We consider the $\lambda$-divergence $\Dren[\lambda]{p}{q}\ldef \En_{q}\prn[\big]{\frac{dp}{dq}}^\lambda-1$. Then, as long as $\lambda(\lambda-1)r^2 \eta<\sigma^2$, we can calculate
\begin{align*}
    2\log\prn*{1+\Dren[\lambda]{\Qr}{\Qr[0]}}
    =&~(\lambda-1)\log \det(\Sigma_0)-\lambda \log\det(\Sigma_r)-\log\det (\lambda \Sigma_r\inv-(\lambda-1)\Sigma_0\inv) \\
    =&~(\lambda-1)\log \det(\Sigma_0)-(\lambda-1)\log\det(\Sigma_r)-\log\det (\lambda \Id-(\lambda-1)\Sigma_r\Sigma_0\inv) \\
    =&~ -d\log\prn[\Big]{1-\frac{\lambda(\lambda-1)r^2\eta}{\sigma^2}}\,.
\end{align*}
In particular, as long as $\lambda^2\leq \frac{\sigma^2}{2\eta}$, it holds that
\begin{align*}
    1+\max_{r\in[0,1]}\Dren[\lambda]{\Qr}{\Qr[0]}\leq \exp\prn[\Big]{\frac{d\lambda^2\eta}{\sigma^2}}\,.
\end{align*}
Now, consider the following distribution of $(X_0,W,Z)$:
\begin{align*}
    \bQr: X_0\sim \bar P\,,\; X\sim \normal{X_0,\sigma^2\Id}\,,\; Z\sim \normal{0,\eta\Id}\,,\; W=X+rZ\,. 
\end{align*}
By the data-processing inequality, we know
\begin{align*}
    1+\Dren[\lambda]{\bQr}{\bQr[0]}\leq \exp\prn[\Big]{\frac{d\lambda^2\eta}{\sigma^2}}\leq \exp\prn[\Big]{\frac{\lambda^2}{K}}\,, \qquad \text{for all}~1\leq \lambda\leq \frac{\sigma}{\sqrt{2\eta}}\,,\; ~r\in[0,1]\,.
\end{align*}
In particular, choosing $\lambda=\frac{K}{2}$ gives $2^{-\lambda}\,(1+\Dren[\lambda]{\bQr}{\bQr[0]})\leq \exp\prn*{-\frac{K}{12}}$. Applying \cref{lem:Renyi-to-cov} completes the proof.

(2) We denote $f=\log P$. By \cref{prop:nabla-p-MGF}, 
for $0\leq \lambda\leq \frac{\sigma^2}{4}$, we can bound
\begin{align*}
    \En_{X\sim P} \exp\prn*{ \lambda\nrm{\nabla f(X)}^2}
    \leq \exp\prn[\Big]{\frac{2d\lambda}{\sigma^2}}\,.
\end{align*}

Next, recalling the definition of $\bQr$, we can write
\begin{align*}
\En_{X\sim P,\, Z\sim \normal{0,\eta\Id}} \exp\prn*{ \lambda\abs*{\tri{Z, \nabla f(X+r Z)}} } 
    =&~ \En_{X_0\sim \bar P} \En_{X\sim \normal{X_0,\sigma^2\Id},\, Z\sim \normal{0,\eta\Id}} \exp\prn*{ \lambda\abs*{\tri{Z, \nabla f(X+r Z)}} } \\
    =&~ \En_{(Z,W)\sim \bQr} \exp\prn*{ \lambda\abs*{\tri{Z, \nabla f(W)}}}\,.
\end{align*}
By the Cauchy--Schwarz inequality,
\begin{align*}
    \En_{(W,Z)\sim \bQr} \exp\prn*{ \lambda\abs*{\tri{Z, \nabla f(W)}} }
    \leq&~ \sqrt{\prn*{1+\Dchis{\bQr}{\bQr[0]}}\En_{(W,Z)\sim \bQr[0]} \exp\prn*{ 2\lambda\abs*{\tri{Z, \nabla f(W)}} } } \\
    \leq&~ e\cdot \sqrt{\En_{W\sim \bQr[0]}[e^{4\lambda^2\eta\nrm{\nabla f(W)}^2}]}\,,
\end{align*}
where we use that under $\bQr[0]$, $\tri{Z, \nabla f(W)}\mid W\sim \normal{0,\eta\nrm{\nabla f(W)}^2}$. Note that the distribution of $W\sim \bQr[0]$ is exactly $P$. Hence, this implies that as long as $8\lambda^2\eta\leq \frac{\sigma^2}{4}$,
\begin{align*}
    \En_{(W,Z)\sim \bQr} \exp\prn*{ \lambda\abs*{\tri{Z, \nabla f(W)}} } \leq \exp\prn[\Big]{1+\frac{8d\lambda^2\eta}{\sigma^2}}\leq \exp\prn[\Big]{1+\frac{8\lambda^2}{K}}\,. 
\end{align*}
Taking $\lambda=\frac{\min\crl{\B,1}}{16}K$ gives
\begin{align*}
    \En_{(W,Z)\sim \bQr} \prn*{ \exp\prn*{ \abs*{\tri{Z, \nabla f(W)}}} - e^\B}_+
    \leq&~ e^{-(\lambda-1)B}\En_{(W,Z)\sim \bQr} \exp\prn*{ \lambda \abs*{\tri{Z, \nabla f(W)}}}  \\
    \leq&~ \exp\prn[\Big]{\B+1-\frac{\min\crl{\B,1}\B}{32}K}\,.
\end{align*}






\end{proof}
